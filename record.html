function render(data){ let table=document.getElementById("table"); table.innerHTML=` 日期 上班 下班 工時 薪資 `; let labels=[]; let salary=[]; let totalH=0; let totalPay=0; let workDays=0; data.forEach(r=>{ // 只計算有上班的 if(!r.start) return; let row=table.insertRow(); row.insertCell(0).innerText=r.date; row.insertCell(1).innerText=r.start || "-"; row.insertCell(2).innerText=r.end || "-"; row.insertCell(3).innerText=r.hours || 0; row.insertCell(4).innerText=r.salary || 0; labels.push(r.date); salary.push(r.salary || 0); totalH+=Number(r.hours || 0); totalPay+=Number(r.salary || 0); workDays++; }); document.getElementById("totalH").innerText=totalH.toFixed(2); document.getElementById("totalPay").innerText=totalPay.toFixed(0); // 👉 新增：上班天數 if(!document.getElementById("workDays")){ let p=document.createElement("p"); p.innerHTML=`本月上班天數：0 天`; document.querySelector(".summary").appendChild(p); } document.getElementById("workDays").innerText=workDays; drawChart(labels,salary); }
